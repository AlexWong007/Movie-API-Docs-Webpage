GET /movies

Get all the movies held in the array of movies


Example Request:

GET http://localhost:4000/movies


Example Response:
[
    {
        "id": 1,
        "movieName": "Inception",
        "description": "A mind-bending science fiction thriller directed by Christopher Nolan.",
        "releaseDate": "2010-07-16"
    },
    {
        "id": 2,
        "movieName": "The Shawshank Redemption",
        "description": "A classic drama film based on Stephen King's novella, directed by Frank Darabont.",
        "releaseDate": "1994-09-23"
    },
    // and more movies (up to 50)
]

Example Guide:
app.get("/movies", async (req, res) => {
    try {
        const response = await axios.get(`${url}/movies`);
        const result = response.data;
        console.log(result);
        res.send(result);
    } catch (error) {
        res.status(404).send(error.message);
    }
});

GET /movies/:id

Get a movie with the specific ID


Example Request:

GET http://localhost:4000/movies/1


Example Response:
{
    "id": 15,
    "movieName": "Casablanca",
    "description": "A romantic drama film directed by Michael Curtiz, set during World War II.",
    "releaseDate": "1942-11-26"
}

Example Guide:
app.get("/movies/:id", async (req, res) => {
    const getMoviesId = req.params.id;
    try {
        const response = await axios.get(`${url}/movies/${getMoviesId}`);
        const result = response.data;
        console.log(result);
        res.send(result);
    } catch (error) {
        res.status(404).send({ error: "Movie not found", message: error.message });
    }
});

POST /post

Post a movie to the end of the movies array


Example Request:

POST http://localhost:4000/post


Request body:
{
    "movie": "Insert movie name here",
    "description": "Insert movie description here",
    "date": "Insert release date of movie here"
}

Example Response:
{
    "id": 51,
    "movieName": "test movie",
    "description": "test description",
    "releaseDate": "0000-00-00"
}

Example Guide:
app.get("/post", async (req, res) => {
    const obj = {
        movie: "test Movie",
        description: "test Description",
        date: "0000-00-00"
    }

    try {
        const response = await axios.post(`${url}/post`, obj);
        console.log(response.data);
        res.send(JSON.stringify(response.data) + " has been posted");
    } catch (error) {
        res.status(404).send({ error: "post has not been posted", message: error.message})
    }
});

PUT /updatePost/:id

Update a movie (you need to update every property)


Example Request:

PUT http://localhost:4000/updatePost/4

Request body:
{
    "movie": "Insert updated movie name here",
    "description": "Insert updated movie description here",
    "date": "Insert updated release date of movie here"
}

Example Response:
{
    "movieName": "Shutter Island",
    "description": "A psychological thriller film directed by Martin Scorsese, known for its suspenseful plot and psychological twists.",
    "releaseDate": "2010-02-19"
}

Example Guide:
app.get("/updatePost/:id", async (req, res) => {
    const getUpdatePostId = req.params.id;
    const obj = {
        movie: "testMovie2",
        description: "testDescription2",
        date: "2222-02-02"
    }

    try {
        const response = await axios.put(`${url}/updatePost/${getUpdatePostId}`, obj);
        console.log(response.data);
        res.send(JSON.stringify(response.data) + " has been updated");
    } catch (error) {
        res.status(404).send({ error: "post has not been updated", message: error.message})
    }
});

PUT /alterPost/:id

Alter a movie (you can either alter all of it, or some of it)


Example Request:

PATCH http://localhost:4000/alterPost/21

Request body:
{
"movieName": "Shutter Island",
"description": "A psychological thriller film directed by Martin Scorsese, known for its suspenseful plot and psychological twists.",
}

Example Response:
{
"movieName": "Shutter Island",
"releaseDate": "2010-02-19"
}

Example Guide:
app.get("/alterPost/:id", async (req, res) => {
    const getAlterPostId = req.params.id;
    const obj = {
        movie: "testMovie3",
    }

    try {
        const response = await axios.patch(`${url}/alterPost/${getAlterPostId}`, obj);
        console.log(response.data);
        res.send(JSON.stringify(response.data) + " has been updated");
    } catch (error) {
        res.status(404).send({ error: "post has not been updated", message: error.message})
    }
});

DELETE /deletePost/:id

Delete a post with a specified ID


Example Request:

DELETE http://localhost:4000/deletedPost/50


Example Response:
{
    "message": "Post deleted"
}

Example Guide:
app.get("/deletePost/:id", async (req, res) => {
    const deleteId = req.params.id;
    try {
        const response = await axios.delete(`${url}/deletePost/${deleteId}`);
        console.log(response.data);
        res.send(response.data);
    } catch (error) {
        res.status(404).send({ error: "post has not deleted", message: error.message})
    }
});

DELETE /deleteAll?key

Delete every single movie that is held in array (this action can only be performed with the key 1VGP2DN-6EWM4SJ-D6FGRHV-Z3PR3TT)


Example Request:

DELETE http://localhost:4000/deleteAll?key=1VGP2DN-6EWM4SJ-D6FGRHV-Z3PR3TT


Example Response:
OK

Example Guide:
const masterKey = "1VGP2DN-6EWM4SJ-D6FGRHV-Z3PR3TT";

app.get("/deleteAll", async (req, res) => {
    try {
        const response = await axios.delete(`${url}/deleteAll?key=${masterKey}`);
        console.log(response.data);
        res.send(response.data);
    } catch (error) {
        res.status(404).send({ error: "all posts have not been deleted", message: error.message})
    }
});
